54 class dl32Singleton
55 {
56 public:
57  typedef T* PointerToInstance;
58  typedef T& ReferenceToInstance;
59  typedef dl32Singleton<T>* RealPointerToInstance;
60 private:

64  static RealPointerToInstance _instance;
65 
66  static void _singleton_instance_deleter() { delete _instance; }
67 protected:
68  dl32Singleton() {}
69  virtual ~dl32Singleton() {} //Correct polymorphic delete
70 public:
76  static ReferenceToInstance instance()
77  {
78  if( _instance == NULLPTR)
79  {
80  /* Register the class at runtime-exit to avoid memory-leaks */
81  atexit( _singleton_instance_deleter );
82 
83  _instance = new T;
84  }
85 
86  return *(static_cast<T*>(_instance));
87  }
