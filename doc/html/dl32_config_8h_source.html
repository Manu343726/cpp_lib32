152 #define DL32FLOAT_EQ(x,y) (((y - DL32FLOAT_EPSILON) < x) && ((y + DL32FLOAT_EPSILON) > x))
153 #define DL32FLOAT_NE(x,y) !DL32FLOAT_EQ(x,y)
154 #define DL32FLOAT_EQCERO(x) ((x < DL32FLOAT_EPSILON) && (x > -DL32FLOAT_EPSILON))
155 #define DL32FLOAT_NECERO(x) !DL32FLOAT_EQCERO(x)
156 
157 #define DL32FLOAT_GE(x,y) (DL32FLOAT_EQ(x,y) || (x > y))
