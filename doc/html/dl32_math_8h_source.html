614  ClippingResult clip( dl32Point2D begin , dl32Point2D end ) const
615  {
616  /* Cohen-Sutherland algorithm */
617 
618  unsigned int begin_code = get_cohen_sutherland_code( begin );
619  unsigned int end_code = get_cohen_sutherland_code( end );
620  unsigned int code;
621  float x , y , dx , dy;
622 
623  while(true)
624  {
625  if( (begin_code | end_code) == COHEN_SUTHERLAND_CODE_CENTER )
626  return ClippingResult(false,begin,end);
627 
628  if( (begin_code & end_code) != COHEN_SUTHERLAND_CODE_CENTER )
629  return ClippingResult(true,begin,end);
630 
631  if( begin_code == COHEN_SUTHERLAND_CODE_CENTER )
632  code = end_code;
633  else
634  code = begin_code;
635 
636  if( code == COHEN_SUTHERLAND_CODE_UP )
637  x = dl32Line2D::get_x_with_y( begin , end , y = Position.y );
638  else if( code == COHEN_SUTHERLAND_CODE_DOWN )
639  x = dl32Line2D::get_x_with_y( begin , end , y = ( Position.y + _height ) );
640  else if( code == COHEN_SUTHERLAND_CODE_LEFT )
641  y = dl32Line2D::get_y_with_x( begin , end , x = Position.x );
642  else if( code == COHEN_SUTHERLAND_CODE_RIGHT )
643  y = dl32Line2D::get_y_with_x( begin , end , x = ( Position.x + _width ) );
644 
645  if( begin_code == code )
646  {
647  begin.x = x;
648  begin.y = y;
649  begin_code = get_cohen_sutherland_code( begin );
650  }
651 
652  if( end_code == code )
653  {
654  end.x = x;
655  end.y = y;
656  end_code = get_cohen_sutherland_code( end );
657  }
658  }
659  }
